package com.prodapt.training.assignments;

public class Employee {

	String eName;
	int eId;
	String eDepartment;
	String eRole;

	Employee() {
		eName = null;
		eId = 0;
		eDepartment = null;
		eRole = null;
	}
	void setData(String eName,int eId,String eDepartment,String eRole) {
		this.eName = eName;
		this.eId = eId;
		this.eDepartment = eDepartment;
		this.eRole = eRole;
	}
	void getData() {
		System.out.println("Employee Name: "+eName);
		System.out.println("Employee Id: "+eId);
		System.out.println("Employee Dept: "+eDepartment);
		System.out.println("Employee Role: "+eRole);
	}
}
